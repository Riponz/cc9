            void BFS(int node, int Mat[20][20], int visit[20])
            {
                int i;
                visit[node] = 1;
                printf("%d ", node);
                count++;
                if(count != size)
                {
                    printf("->");
                }
            
                for(i = 1; i <= size; i++)
                {
                    if(Mat[node][i] && !visit[i])
                    {
                        queue[++front] = i;
                        visit[i] = 1;
                    }
                }
            
                if(count != size)
                {
                    BFS(queue[++rear], Mat, visit);
                }

                }

            int find(int i)
            {
                while(parent[i])
                i=parent[i];
                return i;
            }
            int uni(int i,int j)
            {
                if(i!=j)
                {
                parent[j]=i;
                return 1;
                }
                return 0;
            }

            void floyd(int a[size][size],int n) //floyd function
            {
                int i,j,k;
                for(k=0;k<n;k++)
                {
                  for(i=0;i<n;i++)
                {
                    for(j=0;j<n;j++)
                    {
                        if(a[i][j]>(a[i][k]+a[k][j])) //floyd calculation
                            a[i][j]=a[i][k]+a[k][j];
                    }
                }
            }
            }

                display(n,a);
            
            
                //Apply Warshal's ALgo
                for(k=0;k<n;k++)
                {
                    for(i=0;i<n;i++)

                        for(j=0;j<n;j++)

        #include<stdio.h>
            #define inf 999                        //macro function
            int path[10][10];                      //global variable declaration
            void dijkstra(int n,int s,int cost[10][10],int distance[10])
                                  //function for implementing dijkstra's algo
            {
                     int x,y,u,min,w,set[10],c;
                for(x=1;x<=n;x++)
                                                             //loop for initialization
                {
                    set[x]=0;              //assign initial set matrix with 0
                    distance[x]=cost[s][x];
                    path[x][1]=s;
                    path[x][n]=x;
                }
                c=2;
                while(c<=n)                          //loop for iteration
                {
                    min=99;
                    for(w=1;w<=n;w++)
                                            //loop for setting values of set matrix
                      {
                          if(distance[w]<min && !set[w])
                          {
                              min=distance[w];
                              u=w;
                          }
                      }
                            set[u]=1;
            
            
                        for(w=1;w<=n;w++)
                                   //loop for setting values of distance matrix
                  {
                       if(distance[u]+cost[u][w]< distance[w] && !set[w])
                     {
                      distance[w]=distance[u]+cost[u][w];
                       x=2;
                      for(y=2;y<=n;y++)
                      {
                                    if(path[u][y]!=0 && x<n)
            {
            path[w][x]=path[u][y];
                               x++;
                        }
                       }
                    }
                   }
                    c++;
                }
                      }

            {

                                                               //variable declaration
